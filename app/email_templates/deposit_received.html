import os import uuid from datetime import datetime, timedelta from square.client import Client from app.services.email_service import send_email_from_file, generate_ics_content from app.services.firebase_setup import db def handle_deposit_received(booking: dict): # --- 1. SQUARE SETUP --- client = Client( access_token=os.getenv("SQUARE_ACCESS_TOKEN"), environment="production" ) # --- 2. CALCULATE DUE DATE (2 DAYS BEFORE EVENT) --- event_date_str = booking.get("date") event_date = datetime.strptime(event_date_str, "%Y-%m-%d") due_date = (event_date - timedelta(days=2)).strftime("%Y-%m-%d") # --- 3. CREATE THE BALANCE INVOICE --- remaining_val = booking.get("remaining", 0) remaining_cents = int(float(remaining_val) * 100) invoice_body = { "invoice": { "location_id": os.getenv("SQUARE_LOCATION_ID"), "order_id": booking.get("order_id"), "primary_recipient": { "customer_id": booking.get("square_customer_id") }, "payment_requests": [{ "request_type": "BALANCE", "due_date": due_date, "amount_money": { "amount": remaining_cents, "currency": "USD" } }], "delivery_method": "SHARE_EXTERNALLY", "title": f"Remaining Balance - Event on {event_date_str}" }, "idempotency_key": str(uuid.uuid4()) } pay_link = "https://www.buzzys.org/pay" try: create_result = client.invoices.create_invoice(body=invoice_body) if create_result.is_success(): invoice = create_result.body["invoice"] publish_result = client.invoices.publish_invoice( invoice_id=invoice["id"], body={ "version": invoice["version"], "idempotency_key": str(uuid.uuid4()) } ) if publish_result.is_success(): pay_link = publish_result.body["invoice"]["public_url"] except Exception as e: print(f"SQUARE INVOICE ERROR: {str(e)}") # --- 4. PREPARE ITEM LIST --- items_raw = booking.get("items", []) item_names = [ i.get("title") or i.get("name") or "Party Gear" for i in items_raw if isinstance(i, dict) ] display_items = ", ".join(item_names) if item_names else "Party Gear" # --- 5. PREPARE EMAIL DATA --- ics_content = generate_ics_content(booking) total_val = ( booking.get("pricing_breakdown", {}).get("total") or booking.get("total", 0) ) email_data = { "customer_name": booking.get("name"), "event_date": event_date_str, "total_amount": f"{float(total_val):.2f}", "deposit_amount": f"{float(booking.get('deposit', 0)):.2f}", "remaining_amount": f"{float(remaining_val):.2f}", "booking_id": booking.get("booking_id"), "items": display_items, "pay_link": pay_link, "due_date": due_date } attachments = [{ "content": ics_content, "filename": "event-reminder.ics" }] # --- 6. SEND EMAIL USING FILE TEMPLATE --- email_res = send_email_from_file( to=[booking.get("email")], template_name="deposit_received.html", subject="Your Booking Is Confirmed!", params=email_data ) # --- 7. SAVE EMAIL ID FOR RESEND WEBHOOK --- email_id = email_res.get("response", {}).get("id") if email_id: db.collection("bookings").document(booking["booking_id"]).update({ "last_email_id": email_id, "emailStatus": "Sent" }) return email_res
